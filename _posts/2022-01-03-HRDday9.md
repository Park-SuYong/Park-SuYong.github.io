## 추상클래스, 인터페이스

* 추상클래스
  * 붕어빵 틀들의 틀
  * 객체화가 불가능하다
  * 자식클래스에게 강제성을 부여하기 위해서 
  * 강제성덕분에 실수를 줄여주는 효과 또한 있다
* 만들어보기

```
abstract class Shape{ //추상클래스는 abstract를 붙인다
  
   String name;
   Shape(String name){

      this.name=name;
   } 
   abstract void draw();//바디를 구현하지 않는 추상메서드

}
class Rect extends Shape{//자식클래스
   Rect(String name) {
      super(name);
   }
   @Override // 추상메서드가 갖는 강제성
   void draw() {
      System.out.println("네모 그리기");
   }
}
```

* 인터페이스
  * 추상클래스와 비슷한 역할을 하지만 상속관계는 아니다

```
package class01;

interface UseTV{ //인터페이스 생성
	int MAX=100;//상수다. 올대문자로 쓰기도 한다
	int MIN=0;
	void powerOnOff();
	void soundUp();
	void soundDown();
}

class Remote implements UseTV{ //인터페이스를 불러올떈 implements를 이용한다
	
	TV tv;
	
	Remote(TV tv){
		this.tv = tv;
	}

	@Override //강제성이 부여되기 때문에 사람의 실수를 막아준다
	public void powerOnOff() {
		tv.powerOnOff();
	}

	@Override
	public void soundUp() {
		tv.soundUp();
	}

	@Override
	public void soundDown() {
		tv.soundDown();
	}
}

class TV implements UseTV{
	boolean power;
	int sound;
	TV(){
		this.power = false;
		this.sound = 50;
	}
	@Override
	public void powerOnOff() {
		if(power) {
			power = false;
			System.out.println("전원 오프");
		}
		power = true;
		System.out.println("전원 온");
	}
	@Override
	public void soundUp() {
		if(!power) {
			System.out.println("전원이 오프입니다");
			return;
		}
		sound+=10;
		if(sound>MAX) {
			sound = MAX;
		}
		
	}
	@Override
	public void soundDown() {
		if(!power) {
			System.out.println("전원이 오프입니다");
			return;
		}
		sound-=10;
		if(sound<MIN) {
			sound = MIN;
		}
	}
}
```
